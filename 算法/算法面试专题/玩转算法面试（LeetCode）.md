```python
def moveZeros(arr):
    left, right = 0, 0
    while right != len(arr):
        if arr[right] != 0:
            arr[left] = arr[right]
            left += 1
        right += 1
    while left != len(arr):
        arr[left] = 0
        left += 1
    return arr
```

```python
class Solution:
    def removeDuplicates(self, nums):
        if len(nums) <= 1:
            return len(nums)
        
        i, j = 0, 0
        while j < len(nums):
            if nums[i] != nums[j]:
                i = i + 1
                nums[i] = nums[j]
            j += 1
        return i + 1
```

```python
class Solution(object):
    def sortColors(self, nums):
        count = [0] * 3
        for num in nums:
            count[num] += 1
```

è¿˜å¯ä»¥ä½¿ç”¨ä¸‰è·¯å¿«æ’ï¼Œåªéœ€è¦ä¸€æ¬¡éå†å³å¯å®Œæˆï¼Œè€Œä¸”ç©ºé—´å¤æ‚åº¦ä¸º$O(1)$!

```python
class Solution(object):
    def sortColors(self, nums):
        left, right, ptr = -1, len(nums), 0
        while ptr < right:
            if nums[ptr] == 1:
                ptr += 1
            elif nums[ptr] == 0:
                left += 1
                nums[left], nums[ptr] = nums[ptr], nums[left]
                ptr += 1
            else:
                right -= 1
                nums[ptr], nums[right] = nums[right], nums[ptr]
        return nums
```

```python
class Solution(object):
    def twoSum(self, numbers, target):
        left, right = 0, len(numbers) - 1
        while left < right:
            if numbers[left] + numbers[right] < target:
                left += 1
            elif numbers[left] + numbers[right] > target:
                right -= 1
            else:
                return [left+1, right+1]
                
        return []
```

```python
        s = re.sub("[^a-z0-9]", "" ,s.lower())

        while l < r:
            if s[l] == s[r]:
                l += 1
                r -= 1
            else:
                return False
        return True
```

> ä¸è¦ç»™å¦å¤–çš„æ•°ç»„åˆ†é…é¢å¤–çš„ç©ºé—´ï¼Œä½ å¿…é¡»åŸåœ°ä¿®æ”¹è¾“å…¥æ•°ç»„ã€ä½¿ç”¨ O(1) çš„é¢å¤–ç©ºé—´è§£å†³è¿™ä¸€é—®é¢˜ã€‚

> ä½ å¯ä»¥å‡è®¾æ•°ç»„ä¸­çš„æ‰€æœ‰å­—ç¬¦éƒ½æ˜¯ ASCII ç è¡¨ä¸­çš„å¯æ‰“å°å­—ç¬¦ã€‚


```python
class Solution(object):
    def reverseString(self, s):
        """
        :type s: List[str]
        :rtype: None Do not return anything, modify s in-place instead.
        """
        l, r = 0, len(s) - 1
        while l < r:
            s[l], s[r] = s[r], s[l]
            l, r = l + 1, r - 1
        return s
```

`ç®€å•` `å¯¹æ’æŒ‡é’ˆ`

> ç¼–å†™ä¸€ä¸ªå‡½æ•°ï¼Œä»¥å­—ç¬¦ä¸²ä½œä¸ºè¾“å…¥ï¼Œåè½¬è¯¥å­—ç¬¦ä¸²ä¸­çš„å…ƒéŸ³å­—æ¯ã€‚

```python
class Solution(object):
    def reverseVowels(self, s):
        """
        :type s: str
        :rtype: str
        """
        sp_list = ['a', 'o', 'e', 'i', 'u', 'A', 'E', 'I', 'O', 'U']
        s = list(s)
        l, r = 0, len(s) - 1
        while l < r:
            if s[l] not in sp_list:
                l += 1
            elif s[r] not in sp_list:
                r -= 1
            else:
                s[l], s[r] = s[r], s[l]
                l += 1
                r -= 1
        return ''.join(s)
```

```python
class Solution(object):
    def maxArea(self, height):
        """
        :type height: List[int]
        :rtype: int
        """
        max_area = 0
        l, r = 0, len(height) - 1
        while l < r:
            cur_area = (r - l) * min(height[l], height[r])
            max_area = max(max_area, cur_area)
            if height[l] < height[r]:
                l += 1
            else:
                r -= 1
        return max_area
```
